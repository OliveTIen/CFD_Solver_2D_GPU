#ifndef _BOUNDARY_H_GPU_
#define _BOUNDARY_H_GPU_

#include "Define.h"

namespace GPU {
	// 该类是为了处理周期边界而暂时设计的，以后会有更好的替代方案
	// 例如将周期边界并入inner
	class PeriodicBoundary {
	public:
		long numEdges;
		int* neighborEdge; // 第i个元素表示第i个边的相邻边的编号，-1表示没有相邻边界
	public:
		void alloc(long _numEdges) {
			numEdges = _numEdges;
			neighborEdge = new int[numEdges];
			for (long i = 0; i < numEdges; i++) {
				neighborEdge[i] = -1;
			}
		}
		void free() {
			delete[] neighborEdge;
		}
	};
}

#endif // !_BOUNDARY_H_GPU_
