17 /*
18 DeviceArray，设备内存上的数组。用host初始化，析构时自动释放设备内存。
19 但是要实现计算的话，需要重写运算符，很麻烦，因此将指针设为公有
20 此外，不知道成员函数是否需要加上__device__
21 */
22 template<typename T>
23 class DArray{
24 public:
25 T* ptr = nullptr;
26 public:
27 DArray(const T* v, unsigned n) {
28 cudaMalloc((void**)&ptr, n * sizeof(T));
29 cudaMemcpy(ptr, v, n * sizeof(T), cudaMemcpyHostToDevice);
30 }
31 ~DArray() {
32 cudaFree(ptr);
33 }
34 };
