100 int Quick_Sort_Partition(int* a, int l, int r) {
101 int pivot_0, pivot_1, i, j;
102 pivot_0 = a[l * 2 + 0];
103 pivot_1 = a[l * 2 + 1];
104 i = l;
105 j = r + 1;
106 while (true) {
107 do ++i; while (i <= r && (a[i * 2] < pivot_0 || a[i * 2] == pivot_0 && a[i * 2 + 1] <= pivot_1));
108 do --j; while (a[j * 2] > pivot_0 || a[j * 2] == pivot_0 && a[j * 2 + 1] > pivot_1);
109 if (i >= j) break;
110 Swap(a[i * 2], a[j * 2]);
111 Swap(a[i * 2 + 1], a[j * 2 + 1]);
112 }
113 Swap(a[l * 2 + 0], a[j * 2 + 0]);
114 Swap(a[l * 2 + 1], a[j * 2 + 1]);
115 return j;
116 }
