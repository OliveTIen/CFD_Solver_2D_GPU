10 // 将U数组转换为ruvp数组
11 inline void U2ruvp_host(const myfloat U[4], myfloat ruvp[4], myfloat gamma) {
12 // U: rho,rho_u,rho_v,rho_E 注意E=e+0.5*V^2
13 // ruvp: rho,u,v,p
14 ruvp[0] = U[0];
15 ruvp[1] = U[1] / U[0];
16 ruvp[2] = U[2] / U[0];
17 ruvp[3] = ruvp[0] * (gamma - 1) * (U[3] / U[0] - (ruvp[1] * ruvp[1] + ruvp[2] * ruvp[2]) * 0.5);
18 }

20 inline void ruvp2U_host(const myfloat ruvp[4], myfloat U[4], myfloat gamma) {
21 // ruvp: rho,u,v,p
22 // U: rho,rho_u,rho_v,rho_E
23 U[0] = ruvp[0];
24 U[1] = ruvp[0] * ruvp[1];
25 U[2] = ruvp[0] * ruvp[2];
26 U[3] = ruvp[3] / (gamma - 1) + 0.5 * ruvp[0] * (ruvp[1] * ruvp[1] + ruvp[2] * ruvp[2]);//rhoE
27 }

29 // 三维 守恒量转场变量
30 inline void U2ruvwp_host_3d(const myfloat U[5], myfloat ruvwp[5], myfloat gamma) {
31
32 /*
33 e=Cv*t, Cv:R:Cp=1:(gamma-1):gama => R*t = (gamma-1)*Cv*t = (gamma-1)*e
34 p=rho*R*t => p=rho*(gamma-1)*e = rho*(gamma-1)*(E-0.5*V2)
35 */
36 ruvwp[0] = U[0];// rho
37 ruvwp[1] = U[1] / U[0];// u
38 ruvwp[2] = U[2] / U[0];// v
39 ruvwp[3] = U[3] / U[0];// w
40 myfloat V2 = ruvwp[1] * ruvwp[1] + ruvwp[2] * ruvwp[2] + ruvwp[3] * ruvwp[3];// 速度平方
41 ruvwp[4] = (U[4] / U[0] - 0.5 * V2) * U[0] * (gamma - 1);// p
42 }

44 // 某方向上的马赫数
45 inline myfloat getMach(const myfloat U[4], myfloat nx, myfloat ny, myfloat gamma) {
46 myfloat ruvp[4]{};
47 U2ruvp_host(U, ruvp, gamma);
48 myfloat rho = ruvp[0];
49 myfloat u = ruvp[1];
50 myfloat v = ruvp[2];
51 myfloat p = ruvp[3];
52 myfloat un = u * nx + v * ny;
53 myfloat a2 = gamma * p / rho;
54 myfloat Ma2 = (u * u + v * v) / a2;
55 return sqrt(Ma2);
56 }
