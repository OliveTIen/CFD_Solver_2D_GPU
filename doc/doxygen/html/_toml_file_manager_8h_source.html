58template<typename T_type>
59inline void TomlFileManager::getValueOrigin(std::string key, T_type& value_to_be_changed) {
60
61 bool failed = false;
62 // 如果是bool类型
63 if (std::is_same<T_type, bool>::value) {
64 // 尝试读取字符串'bool'（不带引号）
65 cpptoml::option<bool> valueBool = m_tree->get_qualified_as<bool>(key);
66 if (valueBool) {
67 value_to_be_changed = *valueBool;
68 }
69 // 如果无法读取字符串'bool'，则按int类型读取数字
70 else {
71 cpptoml::option<int> valueInt = m_tree->get_qualified_as<int>(key);
72 if (valueInt) {
73 value_to_be_changed = (bool)*valueInt;// int转换为bool。即使在tomlTree中存储的valueInt是int类型的2或-1，但由于value_to_be_changed是bool类型，因此隐式转换为1
74 }
75 else {
76 failed = true;
77 }
78 }
79 }
80 // cpptoml不识别float格式，应按照double读取
81 else if (std::is_same<T_type, float>::value) {
82 cpptoml::option<double> valueDouble = m_tree->get_qualified_as<double>(key);
83 if (valueDouble) {
84 value_to_be_changed = *valueDouble;
85 }
86 else {
87 failed = true;
88 }
89 }
90 // 如果是其他类型，正常读取
91 else {
92 cpptoml::option<T_type> valueOther = m_tree->get_qualified_as<T_type>(key);
93 if (valueOther) {
94 value_to_be_changed = *valueOther;
95 }
96 else {
97 failed = true;
98 }
99 }
100
101 if (failed) {
102 std::stringstream ss;
103 //ss << "invalid_value_exception @TomlFileManager::getValueOrigin\n";
104 ss << "Parameter \"" << key << "\" gets a wrong value type.\n";
105 LogWriter::logAndPrintError(ss.str());
106
107
108 has_getValueFailed = true;
109 }
110}

114 if (treeContainsKey(key)) {
